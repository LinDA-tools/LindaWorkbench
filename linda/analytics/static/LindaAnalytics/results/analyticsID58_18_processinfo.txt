# Prepare Data 
 myvars <- names(loaded_data) %in% c('rowID','uri'); 
loaded_data <- loaded_data[!myvars]
loaded_data <- na.omit(loaded_data) # listwise deletion of missing
loaded_data <- scale(loaded_data) # standardize variables
# Determine number of clusters 
 wss <- (nrow(loaded_data)-1)*sum(apply(loaded_data,2,var)); 
for (i in 2:15) wss[i] <- sum(kmeans(loaded_data, centers=i)$withinss); 
png(file='/var/www/html/LindaAnalytics/plots/plotid300.png',width=600);
plot(1:15, wss, type='b', xlab='Number of Clusters', ylab='Within groups sum of squares');
dev.off();
